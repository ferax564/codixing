```rust
/// Generate bridge chunks at boundaries between adjacent spans.
///
/// For each pair of adjacent spans (A, B), creates a bridge chunk that spans
/// the overlap region: last `overlap_ratio` fraction of A through the first
/// `overlap_ratio` fraction of B.
fn generate_bridge_chunks(
    spans: &[RawSpan],
    source: &[u8],
    config: &ChunkConfig,
) -> Vec<RawSpan> {
    if spans.len() < 2 {
        return spans.to_vec();
    }

    // Clamp overlap_ratio to a sane range (0.0–0.5).
    let overlap_ratio = config.overlap_ratio.clamp(0.0, 0.5);

    let mut result: Vec<RawSpan> = Vec::with_capacity(spans.len() * 2);

    for i in 0..spans.len() {
        // Always include the original span.
        result.push(spans[i].clone());

        // Generate a bridge between this span and the next.
        if i + 1 < spans.len() {
            let a = &spans[i];
            let b = &spans[i + 1];

            let a_size = a.byte_end - a.byte_start;
            let b_size = b.byte_end - b.byte_start;
            let overlap_bytes =
                ((a_size + b_size) as f32 / 2.0 * overlap_ratio) as usize;

            if overlap_bytes < 4 {
                continue; // Too small to be useful.
            }

            let half = overlap_bytes / 2;
            let bridge_start = a.byte_end.saturating_sub(half);
            let bridge_end = (b.byte_start + half).min(source.len());

            // Clamp to UTF-8 char boundaries.
            let source_str = match std::str::from_utf8(source) {
                Ok(s) => s,
                Err(_) => continue,
            };
            let bridge_start = clamp_to_char_boundary(source_str, bridge_start);
            let bridge_end = clamp_to_char_boundary(source_str, bridge_end);

            if bridge_start >= bridge_end {
                continue;
            }

            let bridge_size = non_ws_chars(&source[bridge_start..bridge_end]);
            if bridge_size < config.min_chars || bridge_size > config.max_chars {
                continue;
            }

            // Inherit the deeper scope chain.
            let scope = if b.scope.len() > a.scope.len() {
                b.scope.clone()
            } else {
                a.scope.clone()
            };

            result.push(RawSpan {
                byte_start: bridge_start,
                byte_end: bridge_end,
                scope,
            });
        }
    }

    // Sort by byte_start so chunks appear in file order.
    result.sort_by_key(|s| s.byte_start);
    result
}

/// Clamp a byte offset to the nearest valid UTF-8 character boundary.
fn clamp_to_char_boundary(s: &str, offset: usize) -> usize {
    if offset >= s.len() {
        return s.len();
    }
    // Walk backward to find a char boundary.
    let mut pos = offset;
    while pos > 0 && !s.is_char_boundary(pos) {
        pos -= 1;
    }
    pos
}
```

//! Git recency map: maps file paths to their last-commit timestamp.

use std::collections::HashMap;
use std::path::Path;
use std::process::Command;

/// Build a recency map from git log, bounded to the last `days` days.
///
/// Returns a map from relative file path to the most recent commit timestamp (Unix epoch seconds).
/// Files not modified within the window are absent from the map.
pub fn build_recency_map(root: &Path, days: u64) -> HashMap<String, i64> {
    let since = format!("--since={days} days ago");
    let output = Command::new("git")
        .args(["log", "--format=%ct", "--name-only", "--diff-filter=ACMR", &since, "HEAD"])
        .current_dir(root)
        .output();

    let output = match output {
        Ok(o) if o.status.success() => o,
        _ => return HashMap::new(),
    };

    let stdout = String::from_utf8_lossy(&output.stdout);
    let mut map: HashMap<String, i64> = HashMap::new();
    let mut current_ts: Option<i64> = None;

    for line in stdout.lines() {
        let trimmed = line.trim();
        if trimmed.is_empty() {
            continue;
        }
        // Lines that are pure digits are timestamps.
        if let Ok(ts) = trimmed.parse::<i64>() {
            current_ts = Some(ts);
        } else if let Some(ts) = current_ts {
            // File path line — record the max (most recent) timestamp.
            map.entry(trimmed.to_string())
                .and_modify(|existing| {
                    if ts > *existing {
                        *existing = ts;
                    }
                })
                .or_insert(ts);
        }
    }

    map
}

/// Mildly boost results from recently-modified files.
///
/// Files modified within the last 180 days get a boost that decays linearly
/// from +10% (today) to +0% (180 days ago). The weight is intentionally mild
/// to avoid overwhelming relevance signals.
pub struct RecencyBoostStage;

impl SearchStage for RecencyBoostStage {
    fn apply(&self, results: &mut Vec<SearchResult>, ctx: &SearchContext<'_>) -> Result<()> {
        let recency_map = match ctx.recency_map {
            Some(m) if !m.is_empty() => m,
            _ => return Ok(()),
        };

        let now = std::time::SystemTime::now()
            .duration_since(std::time::UNIX_EPOCH)
            .map(|d| d.as_secs() as i64)
            .unwrap_or(0);

        let mut boosted = false;
        for r in results.iter_mut() {
            if let Some(&last_commit) = recency_map.get(&r.file_path) {
                let days = (now - last_commit) / 86400;
                let factor = (1.0 - days as f32 / 180.0).max(0.0);
                if factor > 0.0 {
                    r.score *= 1.0 + 0.1 * factor;
                    boosted = true;
                }
            }
        }
        if boosted {
            sort_descending(results);
        }
        Ok(())
    }
}
```

/// Search with multiple query reformulations, fusing results via RRF.
///
/// Each query string is run through `search_first_pass()` independently.
/// Results are progressively fused using Reciprocal Rank Fusion (k=60).
pub fn search_multi(
    &self,
    queries: &[String],
    base_query: &SearchQuery,
) -> Result<Vec<SearchResult>> {
    if queries.is_empty() {
        return Ok(Vec::new());
    }

    let candidate_limit = (base_query.limit * 3).max(30);

    let mut all_results: Vec<Vec<SearchResult>> = Vec::new();
    for q_text in queries {
        let sub_query = SearchQuery {
            query: expand_query(q_text),
            limit: candidate_limit,
            file_filter: base_query.file_filter.clone(),
            strategy: Strategy::Fast,
            token_budget: None,
        };
        if let Ok(results) = self.search_first_pass(&sub_query) {
            if !results.is_empty() {
                all_results.push(results);
            }
        }
    }

    if all_results.is_empty() {
        return Ok(Vec::new());
    }

    // Progressive pairwise RRF fusion (same pattern as search_deep).
    let mut fused = all_results.remove(0);
    for results in &all_results {
        fused = rrf_fuse(&fused, results, 60.0);
    }
    fused.truncate(base_query.limit);
    Ok(fused)
}
```
