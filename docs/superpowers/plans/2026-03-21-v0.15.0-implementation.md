```rust
use tree_sitter::{Node, Tree};

use super::{EntityKind, Language, LanguageSupport, SemanticEntity, node_line_range, node_text};

pub struct BashLanguage;

const ENTITY_KINDS: &[&str] = &["function_definition", "variable_assignment"];

impl LanguageSupport for BashLanguage {
    fn language(&self) -> Language {
        Language::Bash
    }

    fn tree_sitter_language(&self) -> tree_sitter::Language {
        tree_sitter_bash::LANGUAGE.into()
    }

    fn entity_node_kinds(&self) -> &[&str] {
        ENTITY_KINDS
    }

    fn extract_entities(&self, tree: &Tree, source: &[u8]) -> Vec<SemanticEntity> {
        let mut entities = Vec::new();
        collect_entities(&tree.root_node(), source, &[], &mut entities);
        entities
    }

    fn extract_signature(&self, node: &Node, source: &[u8]) -> Option<String> {
        extract_bash_signature(node, source)
    }

    fn extract_doc_comment(&self, node: &Node, source: &[u8]) -> Option<String> {
        super::extract_preceding_comments(node, source, "#")
    }
}

fn collect_entities(
    node: &Node,
    source: &[u8],
    scope: &[String],
    entities: &mut Vec<SemanticEntity>,
) {
    match node.kind() {
        "function_definition" => {
            let name = node
                .child_by_field_name("name")
                .map(|n| node_text(&n, source).to_string())
                .unwrap_or_default();
            entities.push(SemanticEntity {
                kind: EntityKind::Function,
                name,
                signature: extract_bash_signature(node, source),
                doc_comment: super::extract_preceding_comments(node, source, "#"),
                byte_range: node.start_byte()..node.end_byte(),
                line_range: node_line_range(node),
                scope: scope.to_vec(),
            });
        }
        "variable_assignment" => {
            // Only capture exported variables or top-level assignments.
            let name = node
                .child_by_field_name("name")
                .map(|n| node_text(&n, source).to_string())
                .unwrap_or_default();
            if !name.is_empty() {
                // Check if preceded by `export` or `declare -x`.
                let is_exported = node
                    .prev_sibling()
                    .is_some_and(|s| node_text(&s, source).starts_with("export"));
                if is_exported || scope.is_empty() {
                    entities.push(SemanticEntity {
                        kind: EntityKind::Variable,
                        name,
                        signature: Some(node_text(node, source).lines().next().unwrap_or("").to_string()),
                        doc_comment: super::extract_preceding_comments(node, source, "#"),
                        byte_range: node.start_byte()..node.end_byte(),
                        line_range: node_line_range(node),
                        scope: scope.to_vec(),
                    });
                }
            }
        }
        _ => {}
    }

    let mut cursor = node.walk();
    for child in node.children(&mut cursor) {
        collect_entities(&child, source, scope, entities);
    }
}

fn extract_bash_signature(node: &Node, source: &[u8]) -> Option<String> {
    let text = node_text(node, source);
    match node.kind() {
        "function_definition" => {
            // Show up to the opening brace.
            if let Some(brace) = text.find('{') {
                Some(text[..brace].trim().to_string())
            } else {
                Some(text.lines().next().unwrap_or(text).trim().to_string())
            }
        }
        _ => Some(text.lines().next().unwrap_or(text).trim().to_string()),
    }
}
```

```rust
pub struct MatlabLanguage;

const ENTITY_KINDS: &[&str] = &[
    "function_definition",
    "class_definition",
];

impl LanguageSupport for MatlabLanguage {
    fn language(&self) -> Language {

    }

    fn tree_sitter_language(&self) -> tree_sitter::Language {
        tree_sitter_matlab::LANGUAGE.into()
    }

    fn entity_node_kinds(&self) -> &[&str] {
        ENTITY_KINDS
    }

    fn extract_entities(&self, tree: &Tree, source: &[u8]) -> Vec<SemanticEntity> {
        let mut entities = Vec::new();
        collect_entities(&tree.root_node(), source, &[], &mut entities);
        entities
    }

    fn extract_signature(&self, node: &Node, source: &[u8]) -> Option<String> {
        extract_matlab_signature(node, source)
    }

    fn extract_doc_comment(&self, node: &Node, source: &[u8]) -> Option<String> {
        extract_matlab_doc_comment(node, source)
    }
}

fn collect_entities(
    node: &Node,
    source: &[u8],
    scope: &[String],
    entities: &mut Vec<SemanticEntity>,
) {
    match node.kind() {
        "function_definition" => {
            let name = super::find_name_node(node, source).unwrap_or_default();
            entities.push(SemanticEntity {
                kind: EntityKind::Function,
                name: name.clone(),
                signature: extract_matlab_signature(node, source),
                doc_comment: extract_matlab_doc_comment(node, source),
                byte_range: node.start_byte()..node.end_byte(),
                line_range: node_line_range(node),
                scope: scope.to_vec(),
            });
            // Recurse into nested functions.
            let mut child_scope = scope.to_vec();
            if !name.is_empty() {
                child_scope.push(name);
            }
            let mut cursor = node.walk();
            for child in node.children(&mut cursor) {
                collect_entities(&child, source, &child_scope, entities);
            }
            return; // Don't double-recurse.
        }
        "class_definition" => {
            let name = super::find_name_node(node, source).unwrap_or_default();
            entities.push(SemanticEntity {
                kind: EntityKind::Class,
                name: name.clone(),
                signature: extract_matlab_signature(node, source),
                doc_comment: extract_matlab_doc_comment(node, source),
                byte_range: node.start_byte()..node.end_byte(),
                line_range: node_line_range(node),
                scope: scope.to_vec(),
            });
            let mut child_scope = scope.to_vec();
            if !name.is_empty() {
                child_scope.push(name);
            }
            let mut cursor = node.walk();
            for child in node.children(&mut cursor) {
                collect_entities(&child, source, &child_scope, entities);
            }
            return;
        }
        _ => {}
    }

    let mut cursor = node.walk();
    for child in node.children(&mut cursor) {
        collect_entities(&child, source, scope, entities);
    }
}

fn extract_matlab_signature(node: &Node, source: &[u8]) -> Option<String> {
    let text = node_text(node, source);
    // Take the first line (the function/class declaration line).
    Some(text.lines().next().unwrap_or(text).trim().to_string())
}

fn extract_matlab_doc_comment(node: &Node, source: &[u8]) -> Option<String> {
    // Matlab comments start with %.
    let mut comments = Vec::new();
    let mut sibling = node.prev_sibling();
    while let Some(sib) = sibling {
        if sib.kind() == "comment" {
            let text = node_text(&sib, source).trim().to_string();
            comments.push(text);
            sibling = sib.prev_sibling();
        } else {
            break;
        }
    }
    if comments.is_empty() {
        return None;
    }
    comments.reverse();
    let cleaned: Vec<String> = comments
        .iter()
        .map(|c| c.trim_start_matches('%').trim().to_string())
        .collect();
    Some(cleaned.join("\n"))
}
```

```rust
//! Mermaid diagram language support — line-based symbol extraction.
//!
//! Extracts diagram types, subgraph labels, node definitions, and class names.

use super::{ConfigLanguageSupport, EntityKind, Language, SemanticEntity};

pub struct MermaidLanguage;

impl ConfigLanguageSupport for MermaidLanguage {
    fn language(&self) -> Language {
        Language::Mermaid
    }

    fn extract_entities(&self, source: &[u8]) -> Vec<SemanticEntity> {
        let text = String::from_utf8_lossy(source);
        extract_mermaid_entities(&text)
    }
}

/// Diagram type keywords recognized by Mermaid.
const DIAGRAM_TYPES: &[&str] = &[
    "flowchart", "graph", "sequenceDiagram", "classDiagram",
    "stateDiagram", "erDiagram", "gantt", "pie", "gitgraph",
    "mindmap", "timeline", "journey",
];

fn extract_mermaid_entities(text: &str) -> Vec<SemanticEntity> {
    let lines: Vec<&str> = text.lines().collect();
    let mut entities = Vec::new();

    for (i, line) in lines.iter().enumerate() {
        let trimmed = line.trim();
        if trimmed.is_empty() || trimmed.starts_with("%%") {
            continue;
        }

        let byte_start: usize = lines[..i].iter().map(|l| l.len() + 1).sum();
        let byte_end = byte_start + line.len();

        let preceding_comment = if i > 0 {
            let prev = lines[i - 1].trim();
            prev.strip_prefix("%%").map(|c| c.trim().to_string())
        } else {
            None
        };

        // Diagram type declaration (first non-comment line usually).
        for dtype in DIAGRAM_TYPES {
            if trimmed.starts_with(dtype) {
                let title = trimmed[dtype.len()..].trim().to_string();
                let name = if title.is_empty() {
                    dtype.to_string()
                } else {
                    format!("{} {}", dtype, title)
                };
                entities.push(SemanticEntity {
                    kind: EntityKind::Type,
                    name,
                    signature: Some(trimmed.to_string()),
                    doc_comment: preceding_comment.clone(),
                    byte_range: byte_start..byte_end,
                    line_range: i..i + 1,
                    scope: vec![],
                });
                break;
            }
        }

        // Subgraph declarations.
        if trimmed.starts_with("subgraph") {
            let label = trimmed.strip_prefix("subgraph").unwrap().trim().to_string();
            if !label.is_empty() {
                entities.push(SemanticEntity {
                    kind: EntityKind::Module,
                    name: label,
                    signature: Some(trimmed.to_string()),
                    doc_comment: preceding_comment.clone(),
                    byte_range: byte_start..byte_end,
                    line_range: i..i + 1,
                    scope: vec![],
                });
            }
        }

        // Node definitions: id[label], id(label), id{label}, id((label)).
        if let Some(node_entity) = parse_node_definition(trimmed, i, byte_start, byte_end) {
            entities.push(node_entity);
        }

        // Class definitions in classDiagram: `class ClassName`.
        if trimmed.starts_with("class ") {
            let name = trimmed
                .strip_prefix("class ")
                .unwrap()
                .split_whitespace()
                .next()
                .unwrap_or("")
                .trim_end_matches('{');
            if !name.is_empty() {
                entities.push(SemanticEntity {
                    kind: EntityKind::Class,
                    name: name.to_string(),
                    signature: Some(trimmed.to_string()),
                    doc_comment: preceding_comment,
                    byte_range: byte_start..byte_end,
                    line_range: i..i + 1,
                    scope: vec![],
                });
            }
        }
    }

    entities
}

/// Try to parse a node definition like `A[Label]`, `B(Round)`, `C{Decision}`.
fn parse_node_definition(
    line: &str,
    line_idx: usize,
    byte_start: usize,
    byte_end: usize,
) -> Option<SemanticEntity> {
    // Match: identifier followed by bracket.
    let open_brackets = ['[', '(', '{'];
    let trimmed = line.trim();

    // Skip lines that are arrows, keywords, or class defs.
    if trimmed.contains("-->")
        || trimmed.contains("---")
        || trimmed.contains("==>")
        || trimmed.starts_with("subgraph")
        || trimmed.starts_with("class ")
        || trimmed.starts_with("end")
    {
        return None;
    }

    for &bracket in &open_brackets {
        if let Some(pos) = trimmed.find(bracket) {
            let id = trimmed[..pos].trim();
            // Must be a simple identifier.
            if !id.is_empty() && id.chars().all(|c| c.is_alphanumeric() || c == '_') {
                let close = match bracket {
                    '[' => ']',
                    '(' => ')',
                    '{' => '}',
                    _ => unreachable!(),
                };
                let label = if let Some(end) = trimmed[pos + 1..].find(close) {
                    trimmed[pos + 1..pos + 1 + end].to_string()
                } else {
                    id.to_string()
                };
                return Some(SemanticEntity {
                    kind: EntityKind::Variable,
                    name: format!("{}: {}", id, label),
                    signature: Some(trimmed.to_string()),
                    doc_comment: None,
                    byte_range: byte_start..byte_end,
                    line_range: line_idx..line_idx + 1,
                    scope: vec![],
                });
            }
        }
    }

    None
}
```

```rust
//! XML/Draw.io language support — quick-xml-based symbol extraction.
//!
//! Extracts element names, IDs, labels, and Draw.io node values.

use super::{ConfigLanguageSupport, EntityKind, Language, SemanticEntity};

pub struct XmlLanguage;

impl ConfigLanguageSupport for XmlLanguage {
    fn language(&self) -> Language {
        Language::Xml
    }

    fn extract_entities(&self, source: &[u8]) -> Vec<SemanticEntity> {
        extract_xml_entities(source)
    }
}

fn extract_xml_entities(source: &[u8]) -> Vec<SemanticEntity> {
    use quick_xml::events::Event;
    use quick_xml::reader::Reader;

    let mut reader = Reader::from_reader(source);
    let mut entities = Vec::new();
    let mut buf = Vec::new();
    let mut scope: Vec<String> = Vec::new();
    let mut is_drawio = false;

    loop {
        match reader.read_event_into(&mut buf) {
            Ok(Event::Start(ref e)) => {
                let tag_name = String::from_utf8_lossy(e.name().as_ref()).to_string();
                let offset = reader.buffer_position();

                if tag_name == "mxGraphModel" {
                    is_drawio = true;
                }

                let attrs = extract_attrs(e);
                emit_entity(&tag_name, &attrs, is_drawio, offset, &scope, &mut entities);

                // Start events push to scope (End events pop).
                scope.push(tag_name);
            }
            Ok(Event::Empty(ref e)) => {
                let tag_name = String::from_utf8_lossy(e.name().as_ref()).to_string();
                let offset = reader.buffer_position();

                if tag_name == "mxGraphModel" {
                    is_drawio = true;
                }

                let attrs = extract_attrs(e);
                emit_entity(&tag_name, &attrs, is_drawio, offset, &scope, &mut entities);
                // Empty (self-closing) elements do NOT push to scope.
            }
            Ok(Event::End(_)) => {
                scope.pop();
            }
            Ok(Event::Eof) => break,
            Err(_) => break,
            _ => {}
        }
        buf.clear();
    }

    entities
}

struct XmlAttrs {
    id: Option<String>,
    name: Option<String>,
    label: Option<String>,
    value: Option<String>,
}

fn extract_attrs(e: &quick_xml::events::BytesStart) -> XmlAttrs {
    let mut attrs = XmlAttrs { id: None, name: None, label: None, value: None };
    for attr in e.attributes().flatten() {
        let key = String::from_utf8_lossy(attr.key.as_ref()).to_string();
        let val = String::from_utf8_lossy(&attr.value).to_string();
        match key.as_str() {
            "id" => attrs.id = Some(val),
            "name" => attrs.name = Some(val),
            "label" => attrs.label = Some(val),
            "value" => attrs.value = Some(val),
            _ => {}
        }
    }
    attrs
}

fn emit_entity(
    tag_name: &str, attrs: &XmlAttrs, is_drawio: bool, offset: usize,
    scope: &[String], entities: &mut Vec<SemanticEntity>,
) {
    if is_drawio && tag_name == "mxCell" {
        if let Some(ref val) = attrs.value {
            if !val.is_empty() {
                let clean = strip_html_tags(val);
                if !clean.is_empty() {
                    entities.push(SemanticEntity {
                        kind: EntityKind::Variable,
                        name: clean,
                        signature: attrs.id.as_ref().map(|i| format!("mxCell id={}", i)),
                        doc_comment: None,
                        byte_range: offset..offset + 1,
                        line_range: 0..1,
                        scope: scope.to_vec(),
                    });
                }
            }
        }
    } else {
        let display_name = attrs.name.clone()
            .or_else(|| attrs.label.clone())
            .or_else(|| attrs.id.clone())
            .unwrap_or_else(|| tag_name.to_string());
        if attrs.name.is_some() || attrs.label.is_some() || attrs.id.is_some() || scope.len() < 2 {
            entities.push(SemanticEntity {
                kind: if scope.is_empty() { EntityKind::Module } else { EntityKind::Variable },
                name: display_name,
                signature: Some(format!("<{}>", tag_name)),
                doc_comment: None,
                byte_range: offset..offset + 1,
                line_range: 0..1,
                scope: scope.to_vec(),
            });
        }
    }
}

/// Strip HTML tags from a string (Draw.io values often contain HTML).
fn strip_html_tags(input: &str) -> String {
    let mut result = String::new();
    let mut in_tag = false;
    for ch in input.chars() {
        match ch {
            '<' => in_tag = true,
            '>' => in_tag = false,
            _ if !in_tag => result.push(ch),
            _ => {}
        }
    }
    result.trim().to_string()
}
```

```rust
    /// Like [`sync()`](Self::sync), but calls `on_progress` at key milestones.
    ///
    /// Progress messages include: "scanning files", "hashing N files",
    /// "indexing N changes", "persisting".
    pub fn sync_with_progress<F>(&mut self, mut on_progress: F) -> Result<SyncStats>
    where
        F: FnMut(&str) + Send + 'static,
    {
        if self.read_only {
            return Err(CodixingError::ReadOnly);
        }
        use crate::watcher::{ChangeKind, FileChange};
        use std::collections::{HashMap, HashSet};

        on_progress("scanning files");

        let old_hashes: HashMap<std::path::PathBuf, FileHashEntry> = self
            .store
            .load_tree_hashes_v2()
            .unwrap_or_default()
            .into_iter()
            .collect();

        let current_files = super::walk_source_files(&self.config.root, &self.config)?;
        on_progress(&format!("found {} files, checking for changes", current_files.len()));

        let mut changes: Vec<FileChange> = Vec::new();
        let mut seen: HashSet<std::path::PathBuf> = HashSet::new();
        let mut unchanged = 0usize;
        let mut skipped_by_mtime = 0usize;
        let mut current_hashes: Vec<(std::path::PathBuf, FileHashEntry)> = Vec::new();

        for abs_path in &current_files {
            seen.insert(abs_path.clone());
            let metadata = std::fs::metadata(abs_path);
            let (current_mtime, current_size) = match &metadata {
                Ok(m) => (m.modified().ok(), m.len()),
                Err(_) => (None, 0),
            };

            if let Some(cached) = old_hashes.get(abs_path) {
                if !cached.file_might_have_changed(current_mtime, current_size) {
                    unchanged += 1;
                    skipped_by_mtime += 1;
                    current_hashes.push((abs_path.clone(), cached.clone()));
                    continue;
                }
            }

            let content = std::fs::read(abs_path)?;
            let hash = xxhash_rust::xxh3::xxh3_64(&content);
            let entry = FileHashEntry::new(hash, current_mtime, current_size);

            match old_hashes.get(abs_path) {
                Some(cached) if cached.content_hash == hash => {
                    unchanged += 1;
                    current_hashes.push((abs_path.clone(), entry));
                }
                _ => {
                    current_hashes.push((abs_path.clone(), entry));
                    changes.push(FileChange {
                        path: abs_path.clone(),
                        kind: ChangeKind::Modified,
                    });
                }
            }
        }

        for old_path in old_hashes.keys() {
            if !seen.contains(old_path) {
                changes.push(FileChange {
                    path: old_path.clone(),
                    kind: ChangeKind::Removed,
                });
            }
        }

        on_progress(&format!("{} changes detected, indexing", changes.len()));

        // Delegate the actual indexing to the existing apply_changes logic.
        // (This is the same code path as sync() from this point onward.)
        let added = changes.iter().filter(|c| matches!(c.kind, ChangeKind::Modified) && !old_hashes.contains_key(&c.path)).count();
        let modified = changes.iter().filter(|c| matches!(c.kind, ChangeKind::Modified) && old_hashes.contains_key(&c.path)).count();
        let removed = changes.iter().filter(|c| matches!(c.kind, ChangeKind::Removed)).count();

        if !changes.is_empty() {
            self.apply_changes(&changes)?;
            on_progress("persisting index");
            self.store.save_tree_hashes_v2(&current_hashes)?;
            self.save()?;
        } else {
            self.store.save_tree_hashes_v2(&current_hashes)?;
        }

        on_progress("sync complete");

        Ok(SyncStats {
            added,
            modified,
            removed,
            unchanged,
        })
    }
```

```rust
use tokio_stream::wrappers::ReceiverStream;
use std::convert::Infallible;

pub async fn sync_sse_handler(
    State(state): State<AppState>,
) -> Sse<impl tokio_stream::Stream<Item = Result<Event, Infallible>>> {
    let (tx, rx) = tokio::sync::mpsc::channel::<Result<Event, Infallible>>(32);

    tokio::spawn(async move {
        // Acquire the write lock in the async context BEFORE entering
        // the blocking thread. tokio::sync::RwLock does not have
        // blocking_write(), so we must acquire it here.
        let mut engine = state.write().await;
        let tx_progress = tx.clone();

        let result = tokio::task::block_in_place(move || {
            engine.sync_with_progress(move |msg| {
                let _ = tx_progress.blocking_send(Ok(
                    Event::default().event("progress").data(msg.to_string())
                ));
            })
        });

        match result {
            Ok(stats) => {
                let data = serde_json::to_string(&stats).unwrap_or_default();
                let _ = tx.send(Ok(Event::default().event("result").data(data))).await;
            }
            Err(e) => {
                let _ = tx.send(Ok(Event::default().event("error").data(e.to_string()))).await;
            }
        }
    });

    Sse::new(ReceiverStream::new(rx))
}
```

**Note:** We use `block_in_place` (not `spawn_blocking`) because we need to hold the write lock guard across the blocking call. `block_in_place` runs the closure on the current thread, which keeps the guard valid. The `blocking_send` on the progress channel bridges back to the async SSE stream. Requires `tokio` with the `rt-multi-thread` feature (already enabled via `features = ["full"]`).

```rust
            CodixingError::Config(format!(
                "failed to write federation config to {}: {e}",
                path.display()

        })?;
        Ok(())
```

```rust
pub fn call_federation_add_project(args: &Value) -> (String, bool) {
    let config_path = args

        Err(e) => (format!("Failed to save config: {e}"), true),

pub fn call_federation_remove_project(args: &Value) -> (String, bool) {
    let config_path = args
        .get("config")
        .and_then(|v| v.as_str())
        .unwrap_or("codixing-federation.json");
    let name = match args.get("name").and_then(|v| v.as_str()) {
        Some(n) => n,
        None => return ("Missing required `name` parameter".to_string(), true),
    };

    let path = PathBuf::from(config_path);
    let mut cfg = match FederationConfig::load(&path) {
        Ok(c) => c,
        Err(e) => return (format!("Failed to load config: {e}"), true),

    if cfg.projects.len() == before {

    }

    match cfg.save(&path) {
        Ok(()) => (format!("Removed project `{name}`"), false),
        Err(e) => (format!("Failed to save config: {e}"), true),

/// Handle `federation_list`.
pub fn call_federation_list(args: &Value, engine: Option<&codixing_core::federation::FederatedEngine>) -> (String, bool) {
    if let Some(fed) = engine {
        let projects = fed.projects();
        let mut out = format!("## Federation: {} projects\n\n", projects.len());
        for p in &projects {
            let status = if p.loaded { "loaded" } else { "not loaded" };
            out.push_str(&format!(
                "- **{}** — `{}` ({}, {} files)\n",
                p.name,
                p.root.display(),
                status,
                p.file_count,
            ));
        }
        (out, false)
    } else {
        // Fall back to reading config file.
        let config_path = args
            .get("config")
            .and_then(|v| v.as_str())
            .unwrap_or("codixing-federation.json");
        match FederationConfig::load(&PathBuf::from(config_path)) {
            Ok(cfg) => {
                let mut out = format!("## Federation config: {} projects\n\n", cfg.projects.len());
                for p in &cfg.projects {
                    out.push_str(&format!("- `{}` (weight={})\n", p.root.display(), p.weight));
                }
                (out, false)
            }
            Err(e) => (format!("Failed to load config: {e}"), true),
        }
    }
}

/// Handle `federation_search`.
pub fn call_federation_search(args: &Value, engine: Option<&codixing_core::federation::FederatedEngine>) -> (String, bool) {
    let query = match args.get("query").and_then(|v| v.as_str()) {
        Some(q) => q,
        None => return ("Missing required `query` parameter".to_string(), true),

    let limit = args.get("limit").and_then(|v| v.as_u64()).unwrap_or(10) as usize;

    match engine {
        Some(fed) => {
            let sq = codixing_core::SearchQuery::new(query).with_limit(limit);

                Ok(results) => {
                    let mut out = format!("## Federated search: {} results\n\n", results.len());
                    for r in &results {

                            r.project, r.result.file_path, r.result.line_start, r.result.line_end, r.result.score,
                        ));
                    }
                    (out, false)
                }
                Err(e) => (format!("Search failed: {e}"), true),
            }
        }
        None => ("Federation not active. Start with --federation <config.json>".to_string(), true),
    }
}
```
